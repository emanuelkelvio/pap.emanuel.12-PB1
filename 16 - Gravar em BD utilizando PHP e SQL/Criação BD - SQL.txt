CREATE TABLE `bd1`.`livrovisitas` (`id_mensagem` INT(5) NOT NULL AUTO_INCREMENT , `nome` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_general_ci NOT NULL , `mensagem` TEXT CHARACTER SET latin1 COLLATE latin1_general_ci NOT NULL , `data` DATE NOT NULL DEFAULT '0000-00-00' , `hora` TIME NOT NULL DEFAULT '00:00:00' , PRIMARY KEY (`id_mensagem`(5))) ENGINE = InnoDB;
